Python front door

The first Python smoke exercises the scoped local read/query/write path and prints the evidence fields that bound the claim.

Python
from shardloom import context
import shardloom as sl

ctx = context(repo_root="/path/to/shardloom", profile_order=("release", "debug"))

fact = ctx.read_csv("data/fact.csv", schema={
    "id": "int64",
    "group_key": "int64",
    "value": "int64",
    "metric": "float64",
    "flag": "boolean",
})

result = (
    fact.filter(sl.col("flag") == True)
        .select("id", "group_key", "value")
        .limit(1000)
        .collect()
)

print(result.fallback_attempted, result.external_engine_invoked)
